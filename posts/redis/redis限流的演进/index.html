/**
 * 一个简单的限流
 * @param user 用户标识
 * @param action 用户行为
 * @param period 时间窗口
 * @param maxCount 限制次数
 */
public static boolean isAllowed(String user, String action, int period, int maxCount) {
    String key = String.format("ratelimit:%s:%s", user, action);
    long nowTs = System.currentTimeMillis();
    Jedis jedis = IRedisPool.jedisPool.getResource();
    // 针对同一个key的多次操作，通过pipeline节省网络开销，提高执行效率
    Pipeline pipelined = jedis.pipelined();
    pipelined.multi();
    // 记录行为，插入数据
    pipelined.zadd(key, nowTs, "" + nowTs);
    // 移除当前时间窗口之外的数据
    pipelined.zremrangeByScore(key, 0, nowTs - 1000);
    // 获取行为数量
    Response<Long> count = pipelined.zcard(key);
    // 设置过期时间，避免冷客户占据内存，多宽限1秒
    pipelined.expire(key, period+1);
    pipelined.exec();
    pipelined.close();
    // 最后比较在指定时间窗口内是否超过限制次数
    return count.get() <= maxCount;
}
// 使用Demo
public static void main(String[] args) {
    for (int i = 0; i < 2; i++) {
        boolean allowed = isAllowed("johnny", "addPoint", 1000, 1);
        System.out.println(allowed);
    }
}

ZSET限流的缺点

不适合大数据量的限流，例如限定60秒不能超过一百万次，随着请求的增加，会消耗大量内存空间

高级限流之漏斗算法

什么是漏斗算法

假设我们一直往漏斗里面灌水，那么当漏水的频率小于灌水的频率时，很显然漏斗就会被灌满。

这个类似于我们一个服务接口被调用的情况，假设我们有一个扣减积分的接口，我们想要这个接口被同一个客户调用的频率不超过1分钟10次，一旦超过这个频率我们就拒绝服务，对这个服务进行降级，给一个友好提示。

那么，针对漏斗算法来说，我们就可以将这个出水口的漏水频率设置为10次/每分钟，同时给这个漏斗一定的容量——10，这也就意味着用户可以连续提交十次扣减积分的请求，都会将这个请求缓存在漏洞的容量内，以固定的10次每分钟，也就是6秒一个请求的速率进行处理。当用户在6秒内，提交第11次扣减积分请求时，就会触发漏斗的限流阈值，从而实现限流的目的。

static class Funnel {
    int capacity; // 漏斗总容量
    float leakingRate; // 漏水频率
    int leftQuota; // 漏斗剩余容量
    long leakingTs; // 上一次漏水时间

    public Funnel(int capacity, float leakingRate) {
        this.capacity = capacity;
        this.leakingRate = leakingRate;
        this.leftQuota = capacity; // 初始化剩余容量与容量相等
        this.leakingTs = System.currentTimeMillis();
    }

    /**
     * 每次请求前计算容量，腾出剩余容量
     */
    private void makeSpace() {
        long nowTs = System.currentTimeMillis();
        long deltaTs = (nowTs - leakingTs) / 1000;// 秒
        int deltaQuota = (int) (deltaTs * leakingRate);// 计算需要腾出多少容量
        if (deltaQuota < 0) { // 上一次漏水时间过去太久，int溢出，变成负数
            this.leftQuota = capacity;
            this.leakingTs = nowTs;
            return;
        }
        if (deltaQuota < 1) { // 上一次漏水时间太近，不需要腾出空间
            return;
        }
        this.leftQuota += deltaQuota;
        this.leakingTs = nowTs;
        if (this.leftQuota > this.capacity) {
            this.leftQuota = this.capacity; // 剩余空间不能超过最大容量
        }
    }

    private boolean watering(int quota) {
        makeSpace();
        if (this.leftQuota >= quota) { // 判断剩余空间是否足够请求申请
            this.leftQuota -= quota;
            return true;
        }
        return false;
    }

    private Map<String, Funnel> funnels = new HashMap<>();// 每个用户的行为对应一个漏斗

    public boolean isAllowed(String user, String action) {
        String key = String.format("%s:%s", user, action);
        Funnel funnel = funnels.get(key);
        if (funnel == null) {
            funnel = new Funnel(capacity, leakingRate);
            funnels.put(key, funnel);
        }
        return funnel.watering(1);
    }
}

public static void main(String[] args) {
    // 一个60秒内允许操作10次的漏斗
    Funnel funnel = new Funnel(10, 10 / 60);
    // 连续操作11次，第11次会返回false，被限流了
    for (int i = 0; i < 11; i++) {
        boolean allowed = funnel.isAllowed("johnny", "addPoint");
        System.out.println("allowed = " + allowed);
    }
}

需要注意的是

该代码仅用于演示，存在非原子性的问题，也就是如果在多线程的环境下是不安全的，因为在计算剩余空间的时候，需要取值、计算、再赋值。需要适当的加锁来保证限流数据的正确。

Redis中的漏斗算法封装

和布隆过滤器一样，需要额外安装模块Redis-Cell，通过它的封装，提供了具备原子性的限流指令。

可以看到这个模块更加完善，如果当前被限流了，多久之后可以腾出容量允许请求的重试时间都给你算好了，可以自己等待重试时间之后进行重试。

高级限流之令牌桶算法

什么是令牌桶算法

维护一个令牌桶，以固定的频率往里面放置令牌。当请求到来的时候，先去获取令牌，拿到令牌才进行请求处理。

如果没有拿到令牌，就等待重试或者直接拒绝服务。

public static RRateLimiter createLimiter(String key, int count, int period) {
    RRateLimiter rateLimiter = IRedisson.getInstance().getRateLimiter(key);
    // 设置发放令牌的速率
    // overall表示的是全局所有实例的限制，另一个是每个实例单独限制
    rateLimiter.trySetRate(RateType.OVERALL, count, period, RateIntervalUnit.SECONDS);
    return rateLimiter;
}

public static void main(String[] args) throws InterruptedException {
    // 设置以每秒一个的速度往addPoint这个key发放令牌
    RRateLimiter limiter = createLimiter("addPoint", 1, 1);
    ExecutorService pool = Executors.newFixedThreadPool(10);
    for (int i = 0; i < 10; i++) {
        pool.submit(() -> {
            limiter.acquire(1); // 阻塞获取令牌
            System.out.println(LocalDateTime.now() + ":" + Thread.currentThread().getId() + "拿到了令牌");
        });
    }
}

@Override
public RFuture<Boolean> trySetRateAsync(RateType type, long rate, long rateInterval, RateIntervalUnit unit) {
    return commandExecutor.evalWriteAsync(getName(), LongCodec.INSTANCE, RedisCommands.EVAL_BOOLEAN,// 可以看到限流器是通过HASH结构实现的
            "redis.call('hsetnx', KEYS[1], 'rate', ARGV[1]);"
          + "redis.call('hsetnx', KEYS[1], 'interval', ARGV[2]);"
          + "return redis.call('hsetnx', KEYS[1], 'type', ARGV[3]);",
            Collections.<Object>singletonList(getName()),
            rate, unit.toMillis(rateInterval), type.ordinal());
}

private <T> RFuture<T> tryAcquireAsync(RedisCommand<T> command, Long value) {
    return commandExecutor.evalWriteAsync(getName(), LongCodec.INSTANCE, command,
            // 获取限流器的配置，rate令牌桶容量，interval间隔时间，type类型
            "local rate = redis.call('hget', KEYS[1], 'rate');"
          + "local interval = redis.call('hget', KEYS[1], 'interval');"
          + "local type = redis.call('hget', KEYS[1], 'type');"
            // 判断限流器是否已经初始化
          + "assert(rate ~= false and interval ~= false and type ~= false, 'RateLimiter is not initialized')"
          // 全局限制 or 每个实例单独限制
          + "local valueName = KEYS[2];"
          + "if type == '1' then "
              + "valueName = KEYS[3];"
          + "end;"
          // 当前令牌数，第一次是拿不到的，进入else逻辑
          + "local currentValue = redis.call('get', valueName); "
          + "if currentValue ~= false then "
               // 判断令牌数够不够用，不够用，返回时间
                 + "if tonumber(currentValue) < tonumber(ARGV[1]) then "
                     + "return redis.call('pttl', valueName); "
                 + "else "
                  // 够用，减去申请的令牌数
                     + "redis.call('decrby', valueName, ARGV[1]); "
                     + "return nil; "
                 + "end; "
          + "else "
            // 检查申请的令牌是否大于令牌桶的容量
                 + "assert(tonumber(rate) >= tonumber(ARGV[1]), 'Requested permits amount could not exceed defined rate'); "
                 + "redis.call('set', valueName, rate, 'px', interval); "
                 + "redis.call('decrby', valueName, ARGV[1]); "
                 + "return nil; "
          + "end;",
            Arrays.asList(getName(), getValueName(), getClientValueName()),
            value);

另外需要补充的是

令牌桶发放令牌是通过ZSET结构实现的，之前在延时队列的文章《妙用redis实现延时队列》有简单分享过ZSET的使用。

score记录了请求的时间戳，member记录了请求的令牌数。

通过ZSET实现的滑动时间窗口，记录下每个间隔内已发放的令牌数，来获得还能发放的最大令牌数。

也可以通过最后一次请求的时间戳记录，计算出如果令牌数不够，还需要等待多久才能获得令牌。然后通过异步任务进行重试争抢令牌（非公平）。

都是限流，令牌桶与漏斗的差别

漏斗

漏水概率恒定，也就意味着服务器的请求处理速度是固定的。当突发流量时，大量请求可能被拒绝。

令牌桶

只是发放令牌的速度恒定，放行速率不固定（只要桶里有令牌），应对突发流量更灵活。
